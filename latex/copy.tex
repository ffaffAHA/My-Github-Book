    &\text{令}A=\cos(x)\cos(2x)\cdots\cos(2^nx). \\
    &1.若sinx\neq0,添一项sinx,则 \\
    &\sin(x)A \\
    &=\sin(x)\mathrm{cos}(x)\mathrm{cos}(2x)\cdots\mathrm{cos}(2^nx) \\
    &=\frac12\sin(2x)\cos(2x)\cdots\cos(2^nx) \\
    &=\frac1{2^2}\sin(4x)\cdots\cos(2^nx) \\
    &=\frac1{2^{n+1}}\sin\bigl(2^{n+1}x\bigr) \\
    &\text{故}A=\frac{\sin\left(2^{n+1}x\right)}{2^{n+1}\sin x}. \\
    &n\to\infty\text{ 时,}\sin(2^{n+1}x)\text{ 振荡但有界,即}-1\leq\sin(2^{n+1}x)\leq1. \\
    &\bigtriangledown\lim_{n\to\infty}\frac1{2^{n+1}}=0,\text{而}\sin x\neq0\text{ 为常数,} \\
    &\text{故}\lim_{n\to\infty}A=0. \\
